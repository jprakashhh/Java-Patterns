public class Pattern {

	public static void main(String[] args) {
		String s = " ";
		for (int i = 1; i <= 10; i++) {
			for (int j = 10; j > i; j--) {
				System.out.printf("%3s", s);
			}
			for (int k = 1; k <= i; k++) {
				System.out.printf("%3d", k);
			}
			System.out.println();
		}
	}
}
